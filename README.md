# 開発スタート(ルートディレクトリ)
$ docker-compose up -d  

// docker psでコンテナ起動確認  

// コンテナ内に入ってwatch modeで起動

$ docker exec -it nest_crud_api_nest-app-web_1 npm run start:dev   

// localhost:3040でアクセス  


# Installation
$ npm install
Running the app
# development
$ npm run start

# watch mode
$ npm run start:dev

# production mode
$ npm run start:prod
Test
# unit tests
$ npm run test

# e2e tests
$ npm run test:e2e

# test coverage
$ npm run test:cov
